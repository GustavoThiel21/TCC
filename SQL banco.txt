CREATE TABLE IF NOT EXISTS `dados` ( `iddados` int(11) NOT NULL auto_increment, `idusuario` int(11) NOT NULL, `nome` varchar(200) default NULL, `endereco` varchar(200) default NULL, `contato` char(14) default NULL, `email` varchar(200) default NULL, `telcoleira` char(14) default NULL, PRIMARY KEY (`iddados`), KEY `idusuario` (`idusuario`) ) ENGINE=MyISAM DEFAULT CHARSET=latin1 AUTO_INCREMENT=1
